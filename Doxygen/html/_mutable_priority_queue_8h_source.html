104 auto x = H[i];
105 while (i > 1 && *x < *H[vetor(i)]) {

119void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
120 auto x = H[i];
121 while (true) {
122 unsigned k = leftChild(i);
123 if (k >= H.size())
124 break;
125 if (k+1 < H.size() && *H[k+1] < *H[k])
126 ++k; // right child of i
127 if ( ! (*H[k] < *x) )
128 break;
129 set(i, H[k]);
130 i = k;
131 }
132 set(i, x);
133}
