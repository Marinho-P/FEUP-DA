144void Data::deepCopyGraph(Graph<NodeData*>* graph, unordered_map<string,Vertex<NodeData*>*> oldmap, unordered_map<string,Vertex<NodeData*>*> &newmap){
145 for(auto pair:oldmap){
146 auto v = pair.second;
147 NodeData* nodeData;
148 switch (v->getInfo()->getType()) {
149 case CITY: {
150 City *cityNode = (City *) v->getInfo();
151 nodeData = new City(cityNode->getName(), cityNode->getId(), cityNode->getCode(), cityNode->getDemand(),
152 cityNode->getPopulation());
153 break;
154 }
155 case RESERVOIR: {

172 newmap[v->getInfo()->getCode()] = vertex;

174 for(auto pair:oldmap){
175 Vertex<NodeData*>* v = pair.second;
176 for(auto edge: v->getAdj()){

193template <class T>
194void testAndVisit(std::queue< Vertex<T>*> &q, Edge<T> *e, Vertex<T> *w, double residual) {
195 // Check if the vertex 'w' is not visited and there is residual capacity
196 if (! w->isVisited() && residual > 0) {
197 // Mark 'w' as visited, set the path through which it was reached, and enqueue it
198 w->setVisited(true);
199 w->setPath(e);
200 q.push(w);
201 }
202}

213template <class T>
214bool findAugmentingPath(Graph<T> *g, Vertex<T> *s, Vertex<T> *t) {
215 // Mark all vertices as not visited
216 for(auto v : g->getVertexSet()) {
217 v->setVisited(false);
218 }
219 // Mark the source vertex as visited and enqueue it
220 s->setVisited(true);
221 std::queue<Vertex<T> *> q;
222 q.push(s);
223 // BFS to find an augmenting path
224 while (! q.empty() && ! t->isVisited()) {
225 auto v = q.front();
226 q.pop();
227 // Process outgoing edges
228 for(auto e: v->getAdj()) {
229 testAndVisit(q, e, e->getDest(), e->getWeight() - e->getFlow());
230 }
231 // Process incoming edges
232 for(auto e: v->getIncoming()) {
233 testAndVisit(q, e, e->getOrig(), e->getFlow());
234 }
235 }
236 // Return true if a path to the target is found, false otherwise
237 return t->isVisited();
238}

248template <class T>
249double findMinResidualAlongPath(Vertex<T> *s, Vertex<T> *t) {
250 double f = INFINITY;
251 // Traverse the augmenting path to find the minimum residual capacity
252 for (auto v = t; v != s; ) {
253 auto e = v->getPath();
254 if (e->getDest() == v) {
255 f = std::min(f, e->getWeight() - e->getFlow());
256 v = e->getOrig();
257 }
258 else {
259 f = std::min(f, e->getFlow());
260 v = e->getDest();
261 }
262 }
263 // Return the minimum residual capacity
264 return f;
265}

275template <class T>
276void augmentFlowAlongPath(Vertex<T> *s, Vertex<T> *t, double f) {
277 // Traverse the augmenting path and update the flow values accordingly
278 for (auto v = t; v != s; ) {
279 auto e = v->getPath();
280 double flow = e->getFlow();
281 if (e->getDest() == v) {
282 e->setFlow(flow + f);
283 v = e->getOrig();
284 }
285 else {
286 e->setFlow(flow - f);
287 v = e->getDest();
288 }
289 }
290}

300template <class T>
301void edmondsKarp(Graph<T> *g, NodeData* source, NodeData* target) {
302 // Find source and target vertices in the graph
303 Vertex<T>* s = g->findVertex(source);
304 Vertex<T>* t = g->findVertex(target);
305 // Validate source and target vertices
306 if (s == nullptr || t == nullptr || s == t) {
307 throw std::logic_error("Invalid source and/or target vertex");
308 }
309 // While there is an augmenting path, augment the flow along the path
310 while (findAugmentingPath(g, s, t)) {
311 double f = findMinResidualAlongPath(s, t);
312 augmentFlowAlongPath(s, t, f);
313 }
314}

383void Data::print_calculateStatistics(Graph<NodeData*>* graph,double nPipes){
384 double average = 0;
385 double maxdif = 0;
386 double mindif = INFINITY;
387 for(auto v:graph->getVertexSet()){
388 if(v->getInfo()->getCode() == "superSource" || v->getInfo()->getCode() == "superSink"){
389 continue;
390 }
391 for(auto e:v->getAdj()){
392 if(e->getDest()->getInfo()->getCode() == "superSource" || e->getDest()->getInfo()->getCode() == "superSink" ){
393 continue;
394 }
395 double difference = (e->getWeight() - e->getFlow());
396 if(maxdif < difference){
397 maxdif = difference;
398 }
399 if(mindif > difference){
400 mindif = difference;
401 }
402 average+= difference;
403 }
404 }
405 average/=nPipes;
406 double variance = 0;
407 for( auto v: graph->getVertexSet() ){
408 if(v->getInfo()->getCode() == "superSource" || v->getInfo()->getCode() == "superSink"){
409 continue;
410 }
411 for(auto e:v->getAdj()){
412 variance += (e->getFlow() -average)*(e->getFlow()- average);
413 }
414 }
415 variance/= nPipes;
416 cout << "The average difference of capacity-flow is " << average << ", having variance of " << variance << " and max difference " << maxdif << endl;
417}
